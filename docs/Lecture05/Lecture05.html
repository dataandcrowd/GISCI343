Why Write Functions?

Problem: You keep repeating the same code

# Without functions - copy-paste everywhere
sa2 = gpd.read_file('sa2.gpkg')
if sa2.crs != 'EPSG:2193':
    sa2 = sa2.to_crs('EPSG:2193')

parks = gpd.read_file('parks.gpkg')
if parks.crs != 'EPSG:2193':
    parks = parks.to_crs('EPSG:2193')

Solution: Wrap repeated logic in a function

def ensure_nztm(gdf):
    """Reproject to NZTM if needed."""
    if gdf.crs != 'EPSG:2193':
        return gdf.to_crs('EPSG:2193')
    return gdf

Anatomy of a Function

def function_name(parameter1, parameter2="default"):
    """Docstring: what this function does."""
    # Function body
    result = parameter1 + parameter2
    return result
  • def keyword starts the definition
  • Parameters: inputs the function accepts
  • Default values: optional, used when argument not provided
  • Docstring: describes what the function does
  • Return: sends a value back to the caller

Functions with Multiple Returns

def summarise_geodataframe(gdf):
    """Return key summary statistics of a GeoDataFrame."""
    n_features = len(gdf)
    crs = gdf.crs
    bounds = gdf.total_bounds  # [minx, miny, maxx, maxy]
    return n_features, crs, bounds

# Unpack the returned values

Tip

Functions make your code testable, readable, and shareable.

Why Download Data Programmatically?

  • Manual downloads break reproducibility
  • Collaborators cannot re-run your workflow
  • Scripts document exactly what data you used
  • Easy to update when new data is released
  • Essential for open science and sharing via GitHub

Your script is the download step.

Method 2: Export API

The Export API creates a downloadable ZIP archive in three stages:

  1. Create the export job (POST request)
  2. Poll until the job is complete
  3. Download and load the file
import requests
from config import API_KEY

LAYER_ID = 123514
BASE_URL = (
    "https://datafinder.stats.govt.nz"
    "/services/api/v1.x"
)
HEADERS = {
    "Authorization": f"key {API_KEY}",
    "Content-Type": "application/json",
}

Export API: Create and Poll

# Stage 1: Create export job
payload = {
    "crs": "EPSG:2193",
    "formats": {"vector": "application/x-ogc-gpkg"},
    "items": [
        {"item": f"{BASE_URL}/layers/{LAYER_ID}/"}
    ],
}
response = requests.post(
    f"{BASE_URL}/exports/",
    json=payload, headers=HEADERS
)
export_url = response.json()["url"]

# Stage 2: Poll until ready
import time
while True:
    poll = requests.get(export_url, headers=HEADERS)
    state = poll.json()["state"]
    if state == "complete":
        download_url = poll.json()["download_url"]
        break
    time.sleep(5)

Export API: Download and Load

import zipfile, io

# Stage 3: Download and load in memory
dl = requests.get(download_url, headers=HEADERS)

with zipfile.ZipFile(io.BytesIO(dl.content)) as zf:
    gpkg_name = [
        n for n in zf.namelist()
        if n.endswith(".gpkg")
    ][0]
    gdf = gpd.read_file(zf.open(gpkg_name))

print(gdf.shape, gdf.crs)
gdf.plot(figsize=(8, 10))

Note

Use this method when you need a local copy (GeoPackage, Shapefile) or the layer is very large.

Wrapping the API in a Function

def download_statsnz_wfs(layer_id, api_key,
                         crs="EPSG:2193", bbox=None):
    """Download a Stats NZ layer via WFS."""
    wfs_base = (
        f"https://datafinder.stats.govt.nz"
        f"/services;key={api_key}/wfs/"
    )
    params = (
        f"service=WFS&version=2.0.0&request=GetFeature"
        f"&typeNames=layer-{layer_id}"
        f"&outputFormat=application/json"
        f"&srsName={crs}"
    )
    if bbox:
        params += f"&bbox={bbox},{crs}"

    gdf = gpd.read_file(f"{wfs_base}?{params}")
    return gdf

GeoDataFrame Summary Function: Solution

def summarise_gdf(gdf):
    """Return key summary statistics for a GeoDataFrame."""
    if not gdf.crs.is_projected:
        raise ValueError("CRS must be projected for area calculations")

    return {
        "count": len(gdf),
        "total_area_km2": gdf.geometry.area.sum() / 1_000_000,
        "crs": str(gdf.crs),
        "bounds": gdf.total_bounds,
    }

# Try it
import geopandas as gpd

print(f"{stats['count']} zones, {stats['total_area_km2']:.1f} km²")

Common NZ Coordinate Systems

CRS EPSG Units Use Case
NZTM 2193 metres All NZ calculations
WGS84 4326 degrees Web maps, GPS
NZMG 27200 metres Old standard (avoid)

Rule of thumb: Use NZTM (EPSG:2193) for everything in NZ!

Spatial Join Types

Predicates (spatial relationships):

  • 'intersects': Geometries touch or overlap
  • 'within': Left geometry completely inside right
  • 'contains': Left geometry completely contains right
  • 'crosses': Geometries cross each other
# Count sensors per SA2
sensors_per_zone = gpd.sjoin(
    sa2_nztm,
    sensors,
    how='left',
    predicate='contains'
)

sensor_counts = sensors_per_zone.groupby('SA2_name').size()
print(sensor_counts)

Building a Spatial Analysis Toolkit

Combine your API function with analysis functions:

def ensure_nztm(gdf):
    """Reproject to NZTM if needed."""
    if gdf.crs != 'EPSG:2193':
        print(f"Reprojecting from {gdf.crs} to EPSG:2193")
        return gdf.to_crs('EPSG:2193')
    return gdf

def calculate_density(gdf, value_col='population'):
    """Calculate density per km²."""
    gdf = ensure_nztm(gdf)
    gdf['area_km2'] = gdf.geometry.area / 1_000_000
    gdf['density'] = gdf[value_col] / gdf['area_km2']
    return gdf

def create_buffer_analysis(points_gdf, buffer_m=800):
    """Create buffer and return buffered GeoDataFrame."""
    points_gdf = ensure_nztm(points_gdf)
    buffered = points_gdf.copy()
    buffered['geometry'] = points_gdf.geometry.buffer(buffer_m)
    return buffered

Common Pitfalls

1. Wrong CRS for calculations

# ❌ Wrong
distance = point1.distance(point2)  # In WGS84 - wrong!

# ✅ Correct
points = points.to_crs('EPSG:2193')
distance = point1.distance(point2)  # In NZTM - correct!

2. Forgetting to align CRS

# ❌ Wrong - different CRS
result = gpd.overlay(sa2_wgs84, buffer_nztm, how='intersection')

# ✅ Correct - same CRS
sa2_wgs84 = sa2_wgs84.to_crs('EPSG:2193')
result = gpd.overlay(sa2_wgs84, buffer_nztm, how='intersection')

Common Pitfalls (continued)

3. Using wrong spatial predicate

# Points in polygons: use 'within'
points_in_zones = gpd.sjoin(points, zones, predicate='within')

# Overlapping polygons: use 'intersects'
overlapping = gpd.sjoin(zones1, zones2, predicate='intersects')

4. Not handling empty results

# ❌ Wrong
result = gdf[gdf['area_km2'] > 100]
print(result['SA2_name'].iloc[0])  # Crashes if empty!

# ✅ Correct
result = gdf[gdf['area_km2'] > 100]
if len(result) > 0:
    print(result['SA2_name'].iloc[0])
else:
    print("No zones found")

Choropleth Maps with Classification Schemes

A choropleth colours polygons by a data column. The classification scheme controls how values are binned:

# uv pip install mapclassify
import matplotlib.pyplot as plt

sa2["area_km2"] = sa2.geometry.area / 1_000_000

fig, axes = plt.subplots(1, 3, figsize=(18, 8))

for ax, scheme in zip(axes, ["quantiles", "equal_interval", "natural_breaks"]):
    sa2.plot(
        ax=ax,
        column="area_km2",
        scheme=scheme,        # mapclassify handles the binning
        k=5,                  # number of classes
        cmap="YlOrRd",
        legend=True,
        edgecolor="black", linewidth=0.3,
    )
    ax.set_title(scheme.replace("_", " ").title())
    ax.set_axis_off()

plt.tight_layout()

Note

gdf.plot(scheme=...) requires mapclassify. Common schemes: "quantiles" (equal count per bin), "equal_interval" (equal value range), "natural_breaks" (Jenks).

Multi-Scheme Choropleth: Key Ingredients

import matplotlib.pyplot as plt

def compare_classifications(gdf, column, schemes, k=5):
    n = len(schemes)
    fig, axes = plt.subplots(1, n, figsize=(6 * n, 8))
    if n == 1:
        axes = [axes]          # keep iterable for a single scheme

    for ax, scheme in zip(axes, schemes):
        gdf.plot(
            ax=ax, column=column,
            scheme=scheme, k=k,  # <-- mapclassify does the work
            cmap="YlOrRd", legend=True,
            edgecolor="black", linewidth=0.3,
            legend_kwds={"loc": "lower left", "fontsize": 7},
        )
        ax.set_title(scheme.replace("_", " ").title())
        ax.set_axis_off()

    plt.tight_layout()
    return fig

Tip

gdf.plot(scheme=...) delegates to mapclassify under the hood, so pip install mapclassify first.
